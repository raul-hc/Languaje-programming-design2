Reglas Semánticas

programa → definicion:definicion*

defFuncion:definicion → nombre:String parametros:defVariable* tipoRetorno:tipo variableLocales:defVariable* sentencias:sentencia*

esTipoSimple(parametrosi.tipo)

esTipoSimple(tipoRetorno)

sentenciasi.defFuncion = defFuncion

defVariable:definicion → tipo:tipo nombre:String

defStruct:definicion → nombre:String campos:defCampo*

defCampo:definicion → nombre:String tipo:tipo

 

asignacion:sentencia → left:expresion right:expresion

left.tipo == right.tipo

left.modificable == true

esTipoSimple(left.tipo)

escritura:sentencia → exprEscritura:expresion tipoEscritura:String

esTipoSimple(exprEscritura.tipo)

lectura:sentencia → exprLectura:expresion

esTipoSimple(exprLectura.tipo)

exprLectura.modificable == true

return:sentencia → exprRetorno:expresion

exprRetorno.tipo == defFuncion.tipoRetorno

exprRetorno == null ⇒ defFuncion.tipoRetorno==tipoVoid

ifelse:sentencia → condicion:expresion sentenciasIf:sentencia* sentenciasElse:sentencia*

condicion.tipo == tipoEntero

ifelse.sentenciasIfi.definicionFuncion = defFuncion

ifelse.sentenciasElsei.definicionFuncion = defFuncion

while:sentencia → condicion:expresion sentenciasWhile:sentencia*

condicion.tipo == tipoEntero

while.sentenciasi.definicionFuncion = defFuncion

invFuncSent:sentencia → nombreFuncion:String parametros:expresion*

|invFuncSent.parametros| == |invFuncSent.defFuncion.parametros|

invFuncSent.parametrosi.tipo == invFuncSent.defFuncion.parametrosi.tipo

 

expresionBinaria:expresion → left:expresion operador:String right:expresion

left.tipo == tipoEntero OR left.tipo == tipoReal

left.tipo == right.tipo

expresionBinaria.tipo = left.tipo

expresionLogica:expresion → left:expresion operador:String right:expresion

left.tipo == tipoEntero AND right.tipo == tipoEntero

expresionLogica.tipo = left.setTipo/tipoEntero

expresionUnariaNegacion:expresion → expresion:expresion

expresion.tipo == tipoEntero

exprUnariaNegacion.tipo = expresion.tipo/tipoEntero

cast:expresion → tipoDestino:tipo expresionAConvertir:expresion

cast.tipoDestino ≠ cast.expresionAConvertir.tipo

esTipoSimple(tipoDestino)

esTipoSimple(expresionAConvertir.tipo)

expresion.setTipo = TipoDestino

accesoArray:expresion → array:expresion posicion:expresion

accesoArray.posicion.tipo == TipoEntero

accesoArray.array.tipo == TipoArray

accesoArray.tipo = array.tipo

accesoArray.modificable = true

accesoStruct:expresion → struct:expresion nombreCampo:String

accesoStruct.struct.tipo == TipoStruct

accesoStruct.defCampo.nombre[nombreCampo] ≠ ∅

accesoStruct.tipo = struct.tipo.definicionEstructura.campos[nombreCampo]

accesoStruct.modificable = true

invFuncExpr:expresion → nombreFuncion:String parametros:expresion*

|invFuncExpr.parametros| == |invFuncExpr.defFuncion.parametros|

invFuncExpr.parametrosi.tipo == invFuncExpr.defFuncion.parametrosi.tipo

invFuncExpr.defFuncion.tipoRetorno ≠ TipoVoid

invFuncExpr.tipo = invFuncExpr.defFuncion.tipo

variable:expresion → nombre:String

variable.tipo = definicion.tipo

variable.modificable = true

literalInt:expresion → valor:int

literalInt.tipo = tipoEntero

literalReal:expresion → valor:String

literalReal.tipo = tipoReal

literalCaracter:expresion → valor:String

literalCaracter.tipo = tipoCaracter

 

tipoEntero:tipo → λ

tipoReal:tipo → λ

TipoCaracter:tipo → λ

TipoArray:tipo → tamano:int tipo:tipo

tipoIdent:tipo → tipo:String
