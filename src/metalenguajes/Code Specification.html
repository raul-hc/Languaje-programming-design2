run[[programa  →  definicion:definicion* ]] =

 

#source {ficheroEntrada}

CALL MAIN

HALT

define[[defFuncion →nombre:String parametros:defVariable* tipoRetorno:tipo variablesLocales:defVariable* sentencias:sentencia* ]] =

 

#func {nombre}

{nombre}:

#line {start.line}

define[[parametrosi]]

define[[variablesLocalesi]]

ENTER {∑(variablesLocalesi.tipo.size)}

ejecuta[[sentenciasi]]

si tipoRetorno == TipoVoid

   RET 0, ∑{variablesLocalesi.tipo.size}, ∑{parametrosi.tipo.size}

 

 

 

define[[defVariable  →  tipo:tipo  nombre:String ]] =

 

si contexto == ContextoVariable.GLOBAL

   #global {nombre}: {tipoMAPL(tipo)}

si contexto == ContextoVariable.LOCAL

   #local {nombre}: {tipoMAPL(tipo)}

si contexto == ContextoVariable.PARAMETRO

   #param {nombre}: {tipoMAPL(tipo)}

 

 

define[[defStruct  →  nombre:String  campos:defCampo* ]] =

 

#type {nombre} :  

{ 

    define[[camposi]] 

}

 

 

define[[defCampo  →  nombre:String  tipo:tipo ]] =

 

{nombre} : {tipoMAPL(tipo)}

ejecuta[[asignacion  →  left:expresion  right:expresion ]] =

 

#line {end.line}

direccion[[left]]

valor[[right]]

STORE<left.tipo>

 

 

ejecuta[[escritura  →  exprEscritura:expresion  tipoEscritura:String ]] =

 

#line {end.line}
valor[[exprEscritura]]

OUT<exprEscritura.tipo>

 

 si tipoEscritura == printsp

   PUSHB 32

   OUTB

 si tipoEscritura == println

   PUSHB 10

   OUTB

 

 

ejecuta[[lectura  →  exprLectura:expresion ]] =

 

#line {end.line}

direccion[[exprLectura]]

IN<exprLectura.tipo>

STORE< exprLectura.tipo>

 

 

ejecuta[[return  →  exprRetorno:expresion ]] =

#line {end.line}

si valor ≠ null

   valor[[exprRetorno]]

   RET exprRetorno.tipo.size, ∑(defFuncion.variablesLocales), ∑(defFuncion.parametros)

sino

   RET 0, ∑(defFuncion.variablesLocales), ∑(defFuncion.parametros)

 

 

ejecuta[[ifelse  →  condicion:expresion  sentenciasIf:sentencia*  sentenciasElse:sentencia* ]] =

#line {start.line}

valor[[condicion]]

JZ else{n}

valor[[sentenciasIfi]]

JMP finIfElse{n}

else{n}:

valor[[sentenciasElsei]]

finIfElse{n}:

 

 

ejecuta[[while  →  condicion:expresion  sentenciasWhile:sentencia* ]] =

#line {start.line}

inicioWHile{n}:

valor[[condicion]]

JZ finalWhile{n}

valor[[sentenciasWhilei]]

JMP inicioWhile{n}

finalWhile{n}:

 

 

ejecuta[[invFuncSent  →  nombreFuncion:String  parametros:expresion* ]] =

#line {start.line}

valor[[parametrosi]]

CALL {nombreFuncion} :

si invFuncSent.defFuncion.tipoRetorno == TipoVoid

   POP<defFuncion.tipoRetorno>

 

valor [[expresion]]

 

valor[[expresionBinaria  →  left:expresion  operador:String  right:expresion ]] =

 

valor[[left]]

valor[[right]]

                                           // aritméticos

si operador == "+" : ADD<left.tipo>

si operador == "-" : SUB<left.tipo>

si operador == "*" : MUL<left.tipo>

si operador == "/" : DIV<left.tipo>

                                           // relacionales

si operador == ">"  : GT<left.tipo>

si operador == ">=" : GE<left.tipo>

si operador == "<"  : LT<left.tipo>

si operador == "<=" : LE<left.tipo>

 

si operador == "!=" : NE<left.tipo>

si operador == "==" : EQ<left.tipo>

 

valor[[expresionLogica  →  left:expresion  operador:String  right:expresion ]] =

 

valor[[left]]

valor[[right]]

si operador == "&&"

   AND

si operador == "||"

   OR

 

valor[[expresionUnariaNegacion  →  expresion:expresion ]] =

 

valor[[expresion]]

NOT

 

 

valor[[cast  →  tipoDestino:tipo  expresionAConvertir:expresion ]] =

 

si tipoDestino == tipoCaracter && expresionAConvertir.tipo == tipoEntero

   i2b

si tipoDestino == tipoEntero && expresionAConvertir.tipo == tipoCaracter

   b2i

si tipoDestino == tipoEntero && expresionAConvertir.tipo == tipoReal

   f2i

si tipoDestino == tipoReal && expresionAConvertir.tipo == tipoEntero

   i2f

 

 

valor[[accesoArray  →  array:expresion  posicion:expresion ]] =

 

direccion[[accesoArray]]

LOAD<accesoArray.tipo>

 

valor[[accesoStruct  →  struct:expresion  nombreCampo:String ]] =

 

direccion[[accesoStruct]]

LOAD<accesoStruct.defCampo.tipo>

 

valor[[invFuncExpr  →  nombreFuncion:String  parametros:expresion* ]] =

 

#line {end.line}

valor[[parametrosi]]

CALL {nombreFuncion}

 

valor[[variable  →  nombre:String ]] =

 

direccion[[variable]]

LOAD<variable.defVariable.tipo>

 

 

valor[[literalInt  →  valor:int ]] =

 

PUSHi {valor}

 

valor[[literalReal  →  valor:String ]] =

 

PUSHf {valor}

 

valor[[literalCaracter  →  valor:String ]] =

 

PUSHb {valor}

direccion [[expresion]]

 

direccion[[variable  →  nombre:String ]] =

si variable.contexto == ContextoVariable.GLOBAL

   PUSHA {variable.defVariable.direccion}

sino

   PUSH BP

   PUSH {variable.defVariable.direccion}  // Negativa en locales. Positiva en parámetros

   ADD

 

 

direccion[[accesoStruct  →  struct:expresion  nombreCampo:String ]] =

 

direccion[[struct]]

PUSH {accesoStruct.defCampo.direccion}

ADD

 

 

direccion[[accesoArray  →  array:expresion  posicion:expresion ]] =

 

direccion[[array]]

valor[[posicion]]

PUSH {accesoArray.tipo.size}

MUL

ADD

 

 

Nota:

La notación Instrucción<tipo> representa a la versión adecuada de la instrucción para el tipo que se indica:

LOAD<int> → LOADI,  LOAD<float>→ LOADF,  LOAD<char>→ LOADB

tipoMAPL(tipo)

 

Si tipo == TipoReal

     tipoMAPL = float

Si tipo == TipoEntero

     tipoMAPL = int

Si tipo == TipoCaracter

     tipoMAPL = char
